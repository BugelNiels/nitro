258  {

260  // Assumes D65 standard illuminant.
261  const float a11 = 10135552.0f / 24577794.0f;
262  const float a12 = 8788810.0f / 24577794.0f;
263  const float a13 = 4435075.0f / 24577794.0f;
264  const float a21 = 2613072.0f / 12288897.0f;
265  const float a22 = 8788810.0f / 12288897.0f;
266  const float a23 = 887015.0f / 12288897.0f;
267  const float a31 = 1425312.0f / 73733382.0f;

269  const float a33 = 70074185.0f / 73733382.0f;

300  static inline color3 XYZ2CIELab(color3 XYZ, const color3 invReferenceIlluminant = INV_DEFAULT_ILLUMINANT)
301  {
302  const float delta = 6.0f / 29.0f;
303  const float deltaSquare = delta * delta;
304  const float deltaCube = delta * deltaSquare;
305  const float factor = 1.0f / (3.0f * deltaSquare);
306  const float term = 4.0f / 29.0f;
307 
308  // The default illuminant is D65.
309  XYZ = XYZ * invReferenceIlluminant;
310  XYZ.x = (XYZ.x > deltaCube ? powf(XYZ.x, 1.0f / 3.0f) : factor * XYZ.x + term);
311  XYZ.y = (XYZ.y > deltaCube ? powf(XYZ.y, 1.0f / 3.0f) : factor * XYZ.y + term);
312  XYZ.z = (XYZ.z > deltaCube ? powf(XYZ.z, 1.0f / 3.0f) : factor * XYZ.z + term);
313  float L = 116.0f * XYZ.y - 16.0f;
314  float a = 500.0f * (XYZ.x - XYZ.y);
315  float b = 200.0f * (XYZ.y - XYZ.z);
316 
317  return color3(L, a, b);
318  }

320  static inline color3 CIELab2XYZ(color3 Lab, const color3 referenceIlluminant = DEFAULT_ILLUMINANT)
321  {
322  // The default illuminant is D65.
323  float Y = (Lab.x + 16.0f) / 116.0f;
324  float X = Lab.y / 500.0f + Y;
325  float Z = Y - Lab.z / 200.0f;
326 
327  const float delta = 6.0f / 29.0f;
328  const float factor = 3.0f * delta * delta;
329  const float term = 4.0f / 29.0f;
330  X = (X > delta ? X * X * X : (X - term) * factor);
331  Y = (Y > delta ? Y * Y * Y : (Y - term) * factor);
332  Z = (Z > delta ? Z * Z * Z : (Z - term) * factor);
333 
334  return color3(X, Y, Z) * referenceIlluminant;
335  }

372  static inline float HyAB(color3& refPixel, color3& testPixel)
373  {
374  float cityBlockDistanceL = std::fabs(refPixel.x - testPixel.x);
375  float euclideanDistanceAB = std::sqrt((refPixel.y - testPixel.y) * (refPixel.y - testPixel.y) + (refPixel.z - testPixel.z) * (refPixel.z - testPixel.z));

379  static inline float computeMaxDistance(float gqc)
380  {
381  color3 greenLab = color3::XYZ2CIELab(color3::LinearRGB2XYZ(color3(0.0f, 1.0f, 0.0f)));
382  color3 blueLab = color3::XYZ2CIELab(color3::LinearRGB2XYZ(color3(0.0f, 0.0f, 1.0f)));
383  color3 greenLabHunt = color3(greenLab.x, Hunt(greenLab.x, greenLab.y), Hunt(greenLab.x, greenLab.z));
384  color3 blueLabHunt = color3(blueLab.x, Hunt(blueLab.x, blueLab.y), Hunt(blueLab.x, blueLab.z));
385 
386  return powf(HyAB(greenLabHunt, blueLabHunt), gqc);
387  }
